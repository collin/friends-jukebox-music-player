        def run!
          if basic_authentication.provided?
            basic_authentication.authenticate do |login, password|
              user = user_class.authenticate(login, password) 
              unless user
                throw(:halt, lambda{ basic_authentication.request!; "Login Required"} )
              end
              user
            end
          end
        end
